# 第1步：拉取Ubuntu镜像
```
bash
# 在Mac终端执行
docker pull ubuntu:22.04
```
# 第2步：创建并运行容器
```
bash
# 创建共享目录（在Mac上）
mkdir -p ~/dpdk-workspace
```

```
# 运行容器（注意：这里使用 -d 后台运行，方便后续操作）
docker run -d --name dpdk-learning \
  --privileged \
  -v ~/dpdk-workspace:/root/dpdk-workspace \
  -v /dev:/dev \
  -v /sys:/sys \
  --network host \
  ubuntu:22.04 \
  sleep infinity

# 确认容器运行状态
docker ps | grep dpdk-learning
```
# 第3步：进入容器
```
bash
# 进入容器交互式终端
docker exec -it dpdk-learning /bin/bash
```
# 从现在开始的命令都在容器内执行

# 第4步：更新软件包列表
```
bash
# 在容器内执行
apt-get update
```
# 第5步：安装基础依赖包
```
bash
# 安装编译工具和依赖库
apt-get install -y \

  python3-dev \

  libssl-dev \

  vim \

  pciutils
```
# 第6步：下载DPDK源码
```
bash
# 进入/opt目录
cd /opt

# 克隆DPDK源码（使用稳定版本v22.11.2）
git clone https://github.com/DPDK/dpdk.git --branch v22.11.2 --depth 1

# 进入DPDK目录
cd dpdk
```
# 第7步：编译DPDK
```
bash
# 创建构建目录
meson setup build

# 进入构建目录
cd build

# 编译DPDK
ninja

# 安装DPDK
ninja install

# 更新动态链接库缓存
ldconfig
```
# 第8步：设置环境变量
```
bash
# 编辑bashrc文件
echo 'export RTE_SDK=/opt/dpdk' >> ~/.bashrc
echo 'export RTE_TARGET=build' >> ~/.bashrc
echo 'export PATH=$PATH:/opt/dpdk/usertools' >> ~/.bashrc

# 使环境变量生效
source ~/.bashrc
```
# 第9步：验证DPDK安装
```
bash
# 检查环境变量
echo $RTE_SDK
echo $RTE_TARGET

# 查看DPDK版本
cat /opt/dpdk/VERSION

# 查看DPDK工具
ls -la /opt/dpdk/usertools/

# 查看DPDK库文件
ls -la /usr/local/lib/x86_64-linux-gnu/ | grep dpdk
```
# 第10步：编译并运行Hello World示例
```
bash
# 进入helloworld示例目录
cd /opt/dpdk/examples/helloworld

# 编译示例
make

# 运行示例
./build/helloworld
```

# 第11步：创建测试环境（虚拟网卡）
```
bash
# 由于Mac无法使用物理网卡，创建虚拟网卡用于测试

# 创建虚拟网卡对
veth0 和 veth1 是一对"网线"的两端
发送到 veth0 的数据包会自动从 veth1 接收
发送到 veth1 的数据包会自动从 veth0 接收
ip link add veth0 type veth peer name veth1
ip link set veth0 up
ip link set veth1 up

# 创建dummy设备
ip link add dummy0 type dummy
ip link set dummy0 up

# 查看网络设备
ip link show

# 查看DPDK可以识别的设备
/opt/dpdk/usertools/dpdk-devbind.py --status
```
# 第12步：测试DPDK程序开发
```
bash
# 创建自己的测试程序目录
mkdir -p ~/dpdk-workspace/myapp
cd ~/dpdk-workspace/myapp

# 创建简单的DPDK测试程序
cat > hello_dpdk.c << 'EOF'
#include <stdio.h>
#include <rte_eal.h>
#include <rte_debug.h>
#include <rte_version.h>

int main(int argc, char **argv) {
    int ret;
    
    printf("DPDK Hello World Program\n");
    printf("=======================\n");
    
    // 初始化EAL
    ret = rte_eal_init(argc, argv);
    if (ret < 0) {
        rte_panic("Cannot init EAL\n");
    }
    
    printf("DPDK EAL initialized successfully!\n");
    printf("DPDK version: %s\n", rte_version());
    
    // 清理EAL
    rte_eal_cleanup();
    
    printf("Program completed successfully.\n");
    return 0;
}
EOF

# 创建Makefile
cat > Makefile << 'EOF'
# SPDX-License-Identifier: BSD-3-Clause
# Copyright(c) 2010-2014 Intel Corporation

include $(RTE_SDK)/mk/rte.vars.mk

# 应用程序名称
APP = hello_dpdk

# 源文件
SRCS-y := hello_dpdk.c

# 编译选项
CFLAGS += -O3
CFLAGS += $(WERROR_FLAGS)

# 包含DPDK的mk文件
include $(RTE_SDK)/mk/rte.extapp.mk
EOF

# 编译程序
make

# 运行程序（需要EAL参数）
./build/hello_dpdk -l 0 -n 4
第13步：退出和重新进入容器的方法
bash
# 退出容器（在容器内执行）
exit

# 在Mac终端重新进入容器
docker exec -it dpdk-learning /bin/bash

# 如果容器停止了，启动它
docker start dpdk-learning
docker exec -it dpdk-learning /bin/bash
```
# 第14步：清理容器（如果需要重新开始）
```
bash
# 在Mac终端执行
# 停止容器
docker stop dpdk-learning

# 删除容器
docker rm dpdk-learning

# 删除镜像（可选）
docker rmi ubuntu:22.04

# 清理共享目录（可选）
rm -rf ~/dpdk-workspace
```
